    {\normalfont\fontfamily{cmss}\scriptsize
      \ifthenelse{\equal{#1}{}}{}{#1: #2 \hspace{0.7cm}}{}
      #3
      \ifthenelse{\equal{#4}{}}{}{~\nr: #4}
      \ifthenelse{\equal{#5}{}}{}{\vom ~ #5}
      \hspace{0.7cm} - \seite ~ \thepage/\pageref{LastPage} ~- }
}%ende chead
}
